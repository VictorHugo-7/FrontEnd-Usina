    <script>
        const cardContainer = document.getElementById('cardContainer');
        const cards = {
        Loja: [],
        Eventos: [],
        Noticias: [],
        Reportagens: [],
        Trabalhos: []
        };

        // Função para renderizar os cards
        function renderCards() {
        cardContainer.innerHTML = '';
        Object.keys(cards).forEach(type => {
            cards[type].forEach((card, index) => {
            cardContainer.innerHTML += `
                <div class="col">
                <div class="card h-100">
                    <div class="card-body">
                    <h5 class="card-title">${card.titulo} - (${type})</h5>
                    <p class="card-text">${card.descricao}</p>
                    <p class="text-muted">Index: ${index}</p>
                    </div>
                </div>
                </div>`;
            });
        });
        }

        // Adicionar card
        document.getElementById('addForm').addEventListener('submit', function (e) {
        e.preventDefault();
        const type = document.getElementById('addType').value;
        const titulo = document.getElementById('addTitle').value;
        const descricao = document.getElementById('addDescription').value;

        if (titulo && descricao) {
            cards[type].push({ titulo, descricao });
            renderCards();
            alert('Card adicionado com sucesso!');
        } else {
            alert('Preencha todos os campos.');
        }
        });

        // Alterar card
        document.getElementById('editForm').addEventListener('submit', function (e) {
        e.preventDefault();
        const type = document.getElementById('editType').value;
        const index = parseInt(document.getElementById('editIndex').value);
        const titulo = document.getElementById('editTitle').value;
        const descricao = document.getElementById('editDescription').value;

        if (cards[type][index]) {
            cards[type][index] = { titulo, descricao };
            renderCards();
            alert('Card alterado com sucesso!');
        } else {
            alert('Index inválido.');
        }
        });

        // Excluir card
        document.getElementById('deleteForm').addEventListener('submit', function (e) {
        e.preventDefault();
        const type = document.getElementById('deleteType').value;
        const index = parseInt(document.getElementById('deleteIndex').value);

        if (cards[type][index]) {
            cards[type].splice(index, 1);
            renderCards();
            alert('Card excluído com sucesso!');
        } else {
            alert('Index inválido.');
        }
        });

        renderCards(); // Renderiza os cards iniciais (se houver)
    </script>

            // Função para atualizar o card específico
            function atualizarCard() {
                // Pegar o índice do card
                var cardIndex = document.getElementById("cardIndex").value;

                // Verificar se o índice está entre 1 e 4
                if (cardIndex < 1 || cardIndex > 4) {
                    alert("Escolha um número de card entre 1 e 4.");
                    return;
                }

                // Pegar os valores dos inputs
                var arquivoImagem = document.getElementById("imageFile").files[0]; // Pega o arquivo
                var novoTitulo = document.getElementById("title").value;
                var novaDescricao = document.getElementById("description").value;

                // Atualizar a imagem se um arquivo foi selecionado
                if (arquivoImagem) {
                    var leitor = new FileReader(); // Objeto para ler o arquivo
                    leitor.onload = function (event) {
                        // Atualiza a imagem do card selecionado
                        document.getElementById("card-image-" + cardIndex).src = event.target.result;
                    };
                    leitor.readAsDataURL(arquivoImagem); // Lê o arquivo como uma URL de dados
                }

                // Atualizar o título
                if (novoTitulo) {
                    document.getElementById("card-title-" + cardIndex).innerText = novoTitulo;
                }

                // Atualizar o texto
                if (novaDescricao) {
                    document.getElementById("card-text-" + cardIndex).innerText = novaDescricao;
                }

                // Limpar os campos após salvar
                document.getElementById("cardIndex").value = '';
                document.getElementById("imageFile").value = '';
                document.getElementById("title").value = '';
                document.getElementById("description").value = '';
            }
        </script>

            function sortCards() {
                const cardsContainer = document.getElementById('cards-container');
                const cardsArray = Array.from(cardsContainer.children);

                // Função de comparação para ordenar pelos campos de data e horário (formato DD/MM/AAAA HH:MM)
                cardsArray.sort((a, b) => {
                    const dateA = parseDateTime(a.querySelector('.card-date').innerText, a.querySelector('.card-time').innerText);
                    const dateB = parseDateTime(b.querySelector('.card-date').innerText, b.querySelector('.card-time').innerText);
                    return dateA - dateB;
                });

                // Limpa o container e reinsere os cards na nova ordem
                cardsContainer.innerHTML = '';
                cardsArray.forEach((card, index) => {
                    // Atualiza o índice visível no badge
                    card.querySelector('.index-badge').textContent = index + 1;
                    cardsContainer.appendChild(card);
                });
            }

            // Função para converter DD/MM/AAAA e HH:MM em objeto Date
            function parseDateTime(dateStr, timeStr) {
                const [day, month, year] = dateStr.split('/').map(Number); // Converte a string de data
                const [hours, minutes] = timeStr.split(':').map(Number); // Converte a string de horário
                return new Date(year, month - 1, day, hours, minutes); // Cria um objeto Date
            }
